CREATE DEFINER=`root`@`localhost` PROCEDURE `spGetStockTransactionHistory`(
  IN id VARCHAR(240)
)
BEGIN
  #Routine body goes here...
  IF Not EXISTS (SELECT 1 from `transaction` where TransactionID = id) THEN SIGNAL SQLSTATE '45000' SET MESSAGE_TEXT="transaction not present";
  END IF;
  SELECT * from `transaction` where TransactionID = id;
END
